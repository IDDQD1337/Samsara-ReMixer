      Spawn:

actor Disruptor18MMDrop : SamsaraSlotTwoDrop
{
    +DROPPED
    +NEVERRESPAWN
    States
    {
      Spawn:
        SQ72 Q -1
        stop

        TNT1 A 0 A_JumpIfInventory("DisruptorClass",1,"PickupDisruptor")
        goto Super::Pickup

      PickupDisruptor:
        TNT1 A 0 A_JumpIf((CallACS("SamsaraGiveWeapon", 1, 1) & 1) == 1, "PickupSuccess")
        goto PickupFailure
    }
