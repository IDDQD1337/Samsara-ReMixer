			stop
			}

{

	{
		Spawn:
			TK51 A 1
			Loop
		Death:
		Crash:
			TNT1 A 0 A_Playsound("Totenkopf/Ricochet", CHAN_BODY)
			TNT1 A 0 A_JumpIf(CallACS("Totenkopf_Tracer",1,24,48) == 1, 1)
			TK51 BCDE 2 Bright
			Stop

			TNT1 A 0 A_JumpIf(CallACS("Totenkopf_Tracer",1,24,48) == 1, 1)

			TNT1 A 1 Bright
			TNT1 A 0 A_GiveInventory("Totenkopf_BloodGenerator",1,AAPTR_TRACER)
			TNT1 A 1
			Stop
	}
}
