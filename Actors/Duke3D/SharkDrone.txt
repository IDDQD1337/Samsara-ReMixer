	Radius 16
	Height 30
	Speed  4
	PainChance 256
	Scale 0.65

	+FLOORCLIP
	+FLOAT
	+NOGRAVITY
	+NOTDMATCH
	+NOICEDEATH
	+THRUGHOST
	+DONTGIB
	+NOTARGET
	+NOINFIGHTING
	+DONTMORPH
	+NOBLOOD

	PainSound "sharkdrone/pain"
	DeathSound "sentdrone/detonate"
	ActiveSound "sharkdrone/active"
	Tag "Shark Drone"

			SEDS A 0 A_GiveInventory("AlreadySpawned",1)

			TNT1 A 0 A_Recoil(5)

			SEDS A 1 A_Die("Suicide")

			TNT1 A 0 A_SpawnItemEx("SentryDroneBomb",0,0,32,0,0,0,0,1,0)
			TNT1 A 0 A_Stopsound(6)
			TNT1 A 5 A_PlaySound ("sentdrone/detonate", CHAN_AUTO)

	}
