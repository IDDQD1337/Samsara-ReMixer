ACTOR MultiAmmo1Small : Ammo1Small_1 replaces Clip
{
	States
	{
		Spawn:
			TNT1 A 0
			TNT1 A 6 ACS_ExecuteAlways(668,0,3,1)
			TNT1 A 0 A_TakeInventory("MultiplyToken",1)
			TNT1 A 0 A_JumpIfInventory("MultiplyToken",1,"MultiplyAmmo")
			Goto "Super::Spawn"
		MultiplyAmmo:
			TNT1 A 0 A_TakeInventory("MultiplyToken",1)
			TNT1 A 1 A_SpawnItemEx("Ammo1Small_1",0,0,0,random(-1,1),random(-1,1),0,0,40)
			TNT1 A 0 A_JumpIfInventory("MultiplyToken",1,"MultiplyAmmo")
			Goto "Super::Spawn"
	}
}
