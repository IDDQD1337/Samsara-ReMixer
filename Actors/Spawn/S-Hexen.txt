{
	Height 64
	Radius 16
	+BOSSDEATH
	States
	{
		Initialize:
			TNT1 A 0 A_JumpIf(ACS_NamedExecuteWithResult("ReMixerCVarCheck",13) == 0, "See")
			Goto Super::LookWait
		See:
			TNT1 A 0 ACS_NamedExecuteWithResult("ReMixerSpawnScript",2,0,66)
			Goto Super::SeeWait
		Continue:
		BossWait:
			TNT1 A 2 A_JumpIf(CallACS("RemixerInvasionChecker") == 1, "Remove")
			TNT1 A 0 A_ChangeFlag("COUNTKILL",0)
			TNT1 A 0 A_UnsetSolid
			TNT1 A 0 A_UnsetShootable
			TNT1 A 0 A_ChangeFlag("VULNERABLE",1)
			Goto Super::SeeWait
		Death:
			TNT1 A 6 A_BossDeath
			Stop
	}
}

ACTOR KoraxReplacer : ReMixerBaseSpawner replaces Korax
{
	Height 115
	Radius 65
	States
	{
		Initialize:
			TNT1 A 0 A_JumpIf(ACS_NamedExecuteWithResult("ReMixerCVarCheck",13) == 0, "See")
			Goto Super::LookWait
		See:
			TNT1 A 0 ACS_NamedExecuteWithResult("ReMixerSpawnScript",2,0,70)
			Goto Super::SeeWait
		Continue:
		BossWait:
			TNT1 A 2 A_JumpIf(CallACS("RemixerInvasionChecker") == 1, "Remove")
			TNT1 A 0 A_ChangeFlag("COUNTKILL",0)
			TNT1 A 0 A_UnsetSolid
			TNT1 A 0 A_UnsetShootable
			TNT1 A 0 A_ChangeFlag("VULNERABLE",1)
			Goto BossWaitLoop
		BossWaitLoop:
			TNT1 A 1 A_LookEx(0,0,0,0,0,"SeeScript")
			Loop
		SeeScript:
			TNT1 A 0 A_GiveToChildren("KoraxChildInventory",1) //Currently necessitated by Zandronum
			TNT1 A 0 A_GiveToChildren("KoraxChildInventory",1) //Currently necessitated by Zandronum
			Goto Super::SeeWait
		Death:
			TNT1 A 6 ACS_Execute(255)
			Stop
	}
}
