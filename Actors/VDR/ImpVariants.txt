Actor MMRandomizer_VoidImp : DoomImp3
{
    Health 65
    Radius 20
    Height 56
    Speed 7
    PainChance 200
    MONSTER 
    +FLOORCLIP
    SeeSound "imp/sight"
    PainSound "imp/pain"
    DeathSound "imp/death"
    ActiveSound "imp/active"
    Obituary "%o was burned by an imp."
    HitObituary "%o was slashed by an imp."
    States
    {
		Spawn:
			VD15 AB 10 A_Look
			Loop
		See:
			VD15 AABBCCDD 3 A_Chase
			Loop
		Melee:
		Missile:
		    VD15 EF 8 A_FaceTarget
		    TNT1 A 0 A_Jump(128,2)
		    VD15 G 6 A_TroopAttack
		    Goto See
		    VD15 G 6 A_CustomComboAttack("MMRandomizer_LargeImpBall",32,3,"imp/melee")
		    Goto See
		Pain:
		    VD15 H 2
		    VD15 H 2 A_Pain
		    Goto See
		Death:
		    VD15 I 8
		    VD15 J 8 A_Scream
		    VD15 K 6
		    VD15 L 6 A_NoBlocking
		    VD15 M -1
		    Stop
		XDeath:
		    TROO N 5
		    TROO O 5 A_XScream
		    TROO P 5
		    TROO Q 5 A_NoBlocking
		    TROO RST 5
		    TROO U -1
		    Stop
		Raise:
		    VD15 ML 8
		    VD15 KJI 6
		    Goto See
    }
}
