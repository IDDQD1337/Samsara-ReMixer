// Castle Totenkopf SDL
Actor Totenkopf_PistolEnemyDrop : CustomInventory
{
	-INVENTORY.ALWAYSPICKUP
    -COUNTITEM
	+INVENTORY.QUIET
	+DROPPED
	+INVENTORY.NEVERRESPAWN
	//Inventory.PickupMessage "$REMIXER_CLIP_TOTENKOPF"
	Inventory.PickupSound "nosoundforyoubitch"
	States
	{
	Spawn:
		TK02 M -1
		stop
	SpawnCancel:
		TNT1 A 1 A_FadeOut
		stop
	Pickup:
		TNT1 A 0 A_JumpIfInventory("WolfenClassMode", 2, "CheckTotenkopf")
		TNT1 A 0 A_JumpIfInventory("WolfenClass", 1, "CheckNormal")
		TNT1 A 0 A_JumpIfInventory("MarathonClass", 0, "CheckMarathon")
		TNT1 A 0 A_JumpIfInventory("RottClass", 0, "CheckRott")
		TNT1 A 0 A_JumpIfInventory("BondClass", 1, "CheckGoldeneye")
		TNT1 A 0 A_JumpIfInventory("CatacombClass", 0, "CheckCatacomb")
		TNT1 A -1 A_Jump(256,"CheckNormal")
		wait
	CheckNormal:
		TNT1 A 0 A_JumpIfInventory("Clip", 0, "PickupFail")
		goto PickupActual
	CheckTotenkopf:
		TNT1 A 0 A_JumpIfInventory("Clip", 0, 1)
		Goto PickupActual
		TNT1 A 0 A_JumpIfInventory("Totenkopf_IHaveDualPistols", 0, "PickupFail")
		Goto PickupActual
	CheckMarathon:
		TNT1 A 0 A_JumpIfInventory("Clip", 0, 1)
		Goto PickupActual
		TNT1 A 0 A_JumpIfInventory("CanDualPistols", 0, "PickupFail")
		Goto PickupActual
	CheckRott:
		TNT1 A 0 A_JumpIfInventory("Clip", 0, 1)
		Goto PickupActual
		TNT1 A 0 A_JumpIfInventory("Double Pistols", 0, "PickupFail")
		Goto PickupActual
	CheckGoldeneye:
		TNT1 A 0 A_JumpIfInventory("Clip", 0, 1)
		Goto PickupActual
		TNT1 A 0 A_JumpIfInventory("Goldeneye_PP7DualToken", 0, "PickupFail")
		Goto PickupActual
	CheckCatacomb:
		TNT1 A 0 A_JumpIfInventory("Catacomb_Zappers", 0, "PickupFail")
		goto PickupActual
	PickupActual:
		TNT1 A 0 A_JumpIfInventory("WolfenClassMode", 2, "PickupTotenkopf")
		TNT1 A 0 A_JumpIfInventory("WolfenClass", 1, "PickupNormal")
		TNT1 A 0 A_JumpIfInventory("MarathonClass", 0, "PickupMarathon")
		TNT1 A 0 A_JumpIfInventory("RottClass", 0, "PickupRott")
		TNT1 A 0 A_JumpIfInventory("BondClass", 1, "PickupGoldeneye")
		TNT1 A 0 A_JumpIfInventory("CatacombClass", 0, "PickupCatacomb")
		TNT1 A -1 A_Jump(256,"PickupNormal")
		wait
	PickupNormal:
		TNT1 A 0 A_JumpIfInventory("PickupFlashesAreDisabled",1,2)
		TNT1 A 0 A_SetBlend("d7ba45", 0.3, 4)
		TNT1 A 0 A_GiveInventory("Clip", 4)
		TNT1 A -1 A_Jump(256,"PickupSoundNormal")
		wait
	PickupTotenkopf:
		TNT1 A 0 A_JumpIfInventory("PickupFlashesAreDisabled",1,2)
		TNT1 A 0 A_SetBlend("d7ba45", 0.3, 4)
		TNT1 A 0 A_JumpIfInventory("Totenkopf_IHaveDualPistols", 0, 4)
		TNT1 A 0 A_GiveInventory("Totenkopf_IHaveDualPistols", 1)
		TNT1 A 0 A_GiveInventory("Samsara_ModeWeaponChange", 1)
		TNT1 A 0 A_Jump(256,"PickupSoundTotenkopf")
		TNT1 A 0 A_GiveInventory("Clip", 4)
		TNT1 A -1 A_Jump(256,"PickupSoundTotenkopf")
		wait
	PickupMarathon:
		TNT1 A 0 A_JumpIfInventory("PickupFlashesAreDisabled",1,2)
		TNT1 A 0 A_SetBlend("Green", 0.3, 3)
		TNT1 A 0 A_JumpIfInventory("CanDualPistols", 0, 3)
		TNT1 A 0 A_GiveInventory("CanDualPistols", 1)
		TNT1 A 0 A_Jump(256,"PickupSoundNormal")
		TNT1 A 0 A_GiveInventory("Clip", 4)
		TNT1 A -1 A_Jump(256,"PickupSoundNormal")
		wait
	PickupRott:
		TNT1 A 0 A_JumpIfInventory("PickupFlashesAreDisabled",1,2)
		TNT1 A 0 A_SetBlend("d7ba45", 0.3, 4)
		TNT1 A 0 A_JumpIfInventory("Double Pistols", 0, 4)
		TNT1 A 0 A_GiveInventory("Double Pistols", 1)
		TNT1 A 0 A_GiveInventory("GotWeapon2", 1)
		TNT1 A 0 A_Jump(256,"PickupSoundNormal")
		TNT1 A 0 A_GiveInventory("HSMissiles", 3)
		TNT1 A -1 A_Jump(256,"PickupSoundNormal")
		wait
	PickupGoldeneye:
		TNT1 A 0 A_JumpIfInventory("PickupFlashesAreDisabled",1,2)
		TNT1 A 0 A_SetBlend("d7ba45", 0.3, 4)
		TNT1 A 0 A_JumpIfInventory("Goldeneye_PP7DualToken", 0, 3)
		TNT1 A 0 A_GiveInventory("Goldeneye_PP7DualToken", 1)
		TNT1 A 0 A_Jump(256,"PickupSoundNormal")
		TNT1 A 0 A_GiveInventory("Clip", 4)
		TNT1 A -1 A_Jump(256,"PickupSoundNormal")
		wait
	PickupCatacomb:
		TNT1 A 0 A_JumpIfInventory("PickupFlashesAreDisabled",1,2)
		TNT1 A 0 A_SetBlend("d7ba45", 0.3, 4)
		TNT1 A 0 A_GiveInventory("Catacomb_Zappers", 1)
		TNT1 A -1 A_Jump(256,"PickupSoundNormal")
		wait
	PickupSoundNormal:
		TNT1 A 0 ACS_NamedExecuteAlways("ReMixerPickupMessages", 0, 0, 0, 0)
		TNT1 A 1 A_PlaySound("ammo/bulletsmall", CHAN_ITEM)
		stop
	PickupSoundTotenkopf:
		TNT1 A 0 ACS_NamedExecuteAlways("ReMixerPickupMessages", 0, 0, 0, 0)
		TNT1 A 1 A_PlaySound("Totenkopf/getweapon", CHAN_ITEM)
		stop
	PickupFail:
		TK02 M 1
		fail
	}
}
