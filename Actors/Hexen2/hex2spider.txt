Actor Hexen2BlackSpiderLarge
{
	Health 75
	Radius 16
	Height 32
	Mass 100
	Speed 12
	Monster
	BloodType "Hexen2BloodColor"
	BloodColor darkolivegreen
	SeeSound "Hexen2/Spider/See"
	PainSound "Hexen2/Spider/Pain"
	DeathSound "Hexen2/Spider/Death"
	+NODROPOFF
	+FLOORCLIP
	+MISSILEMORE
	States
	{
	Spawn:
		HX01 ABCDEFGHIJKLMNOPQRSTUVWXYZ 3 A_Look
		Loop
	See:
		HX00 AC 2 A_Chase
		HX00 A 0 A_Playsound("Hexen2/Spider/Step", CHAN_BODY)
		HX00 DE 2 A_Chase
		HX00 A 0 A_Playsound("Hexen2/Spider/Step", CHAN_BODY)
		HX00 FG 2 A_Chase
		HX00 A 0 A_Playsound("Hexen2/Spider/Step", CHAN_BODY)
		HX00 HI 2 A_Chase
		HX00 A 0 A_Playsound("Hexen2/Spider/Step", CHAN_BODY)
		HX00 JK 2 A_Chase
		HX00 A 0 A_Playsound("Hexen2/Spider/Step", CHAN_BODY)
		HX00 LM 2 A_Chase
		HX00 A 0 A_Playsound("Hexen2/Spider/Step", CHAN_BODY)
		HX00 NOP 3 A_Chase
		HX00 A 0 A_Playsound("Hexen2/Spider/Step", CHAN_BODY)
		Loop
	Missile:
		HX00 A 0 A_JumpIfCloser(200,"See")
		HX00 A 0 A_Playsound("Hexen2/Spider/Bite", CHAN_BODY)
		HX03 ABC 1 A_FaceTarget
		HX03 E 0 ThrustThingZ(0,Random(30,40),0,0)	
		HX03 E 0 ThrustThing(angle * 256 / 360, Random(10,20), 0, 0)
		HX03 DEFGHIJKLMNOPQRSTU 1 
		HX00 A 0 A_JumpIfCloser(120,"See")
		HX03 I 0 A_CustomMeleeAttack((10),none,none)
		Goto See
	Melee:
		HX00 A 0 A_Playsound("Hexen2/Spider/Bite", CHAN_BODY) 
		TNT1 A 0 A_Jump(128,"Bite2")
	Bite1:
		HX02 A 0 A_FaceTarget
		HX02 ABCDEFGHIJKLMNOPQRS 1
		HX02 S 0 A_CustomMeleeAttack(random(1,9),none,none)
		HX02 TUVW 1
		HX02 W 0 A_FaceTarget
		Goto See
	Bite2:
		HXO4 A 0 A_FaceTarget
		HX04 ABCDEFGHIJKLM 1
		HX04 M 0 A_CustomMeleeAttack(random(1,9),none,none)
		HX04 M 0 A_FaceTarget
		Goto See
	Death.Hexen2Damage:
	Death.Hexen2Fire:
	Death.Hexen2Electric:	
		TNT1 A 0 A_JumpIfHealthLower(-50,"XDeathForDemoness")
		Goto Death				
	Death:
		TNT1 A 0 ACS_NamedExecuteAlways("Samsara_KillCount", 0, 0)
		HX01 A 0 A_NoBlocking
		HX00 A 0 A_Playsound("Hexen2/Spider/Death", CHAN_BODY)
		TNT1 A 0 A_SpawnItemEx("Hexen2BlackSpiderLargeDeath",0,0,0,velx, vely, velz, 0, SXF_NOCHECKPOSITION|SXF_SETMASTER|SXF_TRANSFERSCALE|SXF_ABSOLUTEVELOCITY)		
		TNT1 A 5
		Stop
	Death.Hexen2Extreme:
	Death.Hexen2ElectricExtreme:
	XDeathForDemoness:	
	XDeath:
		TNT1 A 0 A_Playsound("Hexen2/Gib", CHAN_5)
		TNT1 A 0 ACS_NamedExecuteAlways("Samsara_KillCount", 0, 0)  
		TNT1 A 0 A_SpawnItemEx("Hexen2GibSpawnerColor",0,0,0,0,0,0,0,SXF_USEBLOODCOLOR|SXF_NOCHECKPOSITION|SXF_SETMASTER)
		TNT1 A 0 A_NoBlocking
		TNT1 A 5
		stop					
		}
}
